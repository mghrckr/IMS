SELECT 
    BORROW_TRANSACTION.BORROW_DATE,
    BOOK.TITLE AS BOOK_TITLE,
    BORROWER.NAME AS BORROWER_NAME,
    BORROW_TRANSACTION.RETURN_DATE,
    BORROW_TRANSACTION.BOOK_CODE || BORROWER.KTP_NO AS SPECIAL_CODE
FROM 
    BORROW_TRANSACTION
JOIN 
    BOOK ON BORROW_TRANSACTION.BOOK_CODE = BOOK.BOOK_CODE
JOIN 
    BORROWER ON BORROW_TRANSACTION.BORROWER_ID = BORROWER.BORROWER_ID
ORDER BY 
    BORROW_TRANSACTION.BORROW_DATE DESC;
